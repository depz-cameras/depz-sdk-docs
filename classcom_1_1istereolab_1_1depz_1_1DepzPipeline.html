Detailed Description

High-level wrapper over the C-ABI.

depz_pipeline
depz_pipeline
struct depz_pipeline depz_pipeline
Definition: depz_c_api.h:175

: camera → rectify, optionally inference → disparity/depth. The native source / calibration / inference objects it depends on are owned by the pipeline and released, in the correct order, by close().

try (DepzPipeline pipe = DepzPipeline.builder()


        .camera("/dev/video0", 1280, 480, 20)


        .calibrationFromFile(Path.of("calibration_stereo.json"))


        .inference("lightstereo", modelPath, null)   // null → recommended engine


        .enableRectified(true).enableDisparity(true).enableDepth(false)


        .build()) {


    pipe.start();


    try (DepzFrame f = pipe.get(5.0)) {


        DepzImage disp = f.image(Stream.DISPARITY);


    }


    pipe.stop();


}
